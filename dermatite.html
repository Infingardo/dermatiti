                "Scleredema",

                "Scleromixedema",

                "Istiocitosi rare",

                "Dermatite bollosa autoimmune"

            ]

        };


        function calcolaPunteggio() {

            // Calcola il punteggio totale

            const parametri = [

                'spongiosi', 'exocitosi', 'infiltrato', 'eosinofili',

                'neutrofili', 'paracheratosi', 'acantosi', 'atrofia',

                'pigmento', 'mucina', 'ulcerazione', 'distribuzione',

                'prurito', 'velocita'

            ];


            let totale = parametri.reduce((acc, param) => {

                return acc + parseInt(document.getElementById(param).value);

            }, 0);


            // Determina la categoria

            let categoria;

            if(totale <= 15) {

                categoria = "Basso (0-15)";

            } else if(totale <= 30) {

                categoria = "Medio (16-30)";

            } else if(totale <= 42) {

                categoria = "Alto (31-42)";

            } else {

                categoria = "Molto alto (43+)";

            }


            // Mostra i risultati

            const resultDiv = document.getElementById('result');

            const listaDiagnosi = document.getElementById('diagnosi-list');

            

            document.getElementById('punteggio-totale').textContent = `Punteggio Totale: ${totale} (${categoria})`;

            

            listaDiagnosi.innerHTML = diagnosi[categoria]

                .map(d => `<li>${d}</li>`)

                .join('');


            resultDiv.style.display = 'block';

        }

    </script>
